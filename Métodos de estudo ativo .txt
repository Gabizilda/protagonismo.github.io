    function showTab(event, tabId) {
        // Esconder conteúdos
        document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
        // Mostrar selecionado
        document.getElementById(tabId).classList.add('active');
        
        // Resetar botões
        document.querySelectorAll('.tab-btn').forEach(btn => {
            btn.classList.remove('btn-active');
            btn.classList.add('bg-white/5', 'text-white/50');
        });
        
        // Ativar botão clicado
        event.currentTarget.classList.add('btn-active');
        event.currentTarget.classList.remove('bg-white/5', 'text-white/50');
    }

    // Inicializar primeira aba como ativa visualmente
    window.onload = () => {
        const firstBtn = document.querySelector('.tab-btn');
        if(firstBtn) firstBtn.classList.add('btn-active');
    }

        function showTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
            document.getElementById(tabId).classList.add('active');
            
            document.querySelectorAll('.tab-btn').forEach(btn => {
                btn.classList.remove('bg-blue-600', 'text-white', 'shadow-xl');
                btn.classList.add('bg-white/5', 'text-white/50');
            });
            event.currentTarget.classList.add('bg-blue-600', 'text-white', 'shadow-xl');
            event.currentTarget.classList.remove('bg-white/5', 'text-white/50');
        }

        function generateFinalQR() {
            const container = document.getElementById("qrcodeDisplay");
            const url = document.getElementById("qrInput").value;
            if(!url) return;
            container.innerHTML = "";
            container.classList.remove("hidden");
            new QRCode(container, { text: url, width: 180, height: 180 });
        }
    </script>
